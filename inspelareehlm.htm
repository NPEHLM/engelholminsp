    <script>

        // DOM Elements

        const recordButton = document.getElementById('recordButton');

        const pauseButton = document.getElementById('pauseButton');

        const stopButton = document.getElementById('stopButton');

        const timerElement = document.getElementById('timer');

        const statusElement = document.getElementById('status');

        const audioPlayer = document.getElementById('audioPlayer');

        const editorElement = document.getElementById('editor');

        const loaderElement = document.getElementById('loader');

        const downloadButton = document.getElementById('downloadButton');

        const durationDisplay = document.getElementById('recordingDuration');

       

        let mediaRecorder, audioChunks = [], audioBlob, originalAudioBuffer;

        const audioContext = new (window.AudioContext || window.webkitAudioContext)();

        let mediaStream;

        let isRecording = false, isPaused = false, timerInterval, seconds = 0;

 

        recordButton.addEventListener('click', async () => {

            if (!isRecording) {

                try {

                    mediaStream = await navigator.mediaDevices.getUserMedia({ audio: true, video: false });

                    startRecording(mediaStream);

                } catch (err) {

                    console.error("Error accessing microphone:", err);

                    statusElement.textContent = "Kunde inte komma åt mikrofonen.";

                }

            }

        });

 

        pauseButton.addEventListener('click', () => {

            if (isRecording && !isPaused) {

                mediaRecorder.pause();

                isPaused = true;

                clearInterval(timerInterval);

                pauseButton.textContent = `Återuppta`;

                statusElement.textContent = "Inspelning pausad";

            } else if (isRecording && isPaused) {

                mediaRecorder.resume();

                isPaused = false;

                startTimer();

                pauseButton.textContent = `Pausa`;

                statusElement.textContent = "Inspelning pågår...";

            }

        });

 

        stopButton.addEventListener('click', () => {

            if (isRecording) mediaRecorder.stop();

        });

 

        function startRecording(stream) {

            audioChunks = [];

            mediaRecorder = new MediaRecorder(stream);

            mediaRecorder.ondataavailable = e => audioChunks.push(e.data);

            mediaRecorder.onstop = processAudio;

            mediaRecorder.start();

            isRecording = true;

            isPaused = false;

            recordButton.classList.add('hidden');

            [pauseButton, stopButton].forEach(el => el.classList.remove('hidden'));

            editorElement.classList.add('hidden');

            statusElement.classList.remove('hidden');

            statusElement.textContent = "Inspelning pågår...";

            resetTimer();

            startTimer();

        }

 

        async function processAudio() {

            isRecording = false;

            clearInterval(timerInterval);

            if (mediaStream) mediaStream.getTracks().forEach(t => t.stop());

 

            loaderElement.style.display = 'flex';

            statusElement.classList.add('hidden');

            [recordButton, pauseButton, stopButton].forEach(el => el.classList.add('hidden'));

 

            const blobMimeType = mediaRecorder.mimeType || 'audio/webm';

            audioBlob = new Blob(audioChunks, { type: blobMimeType });

            const audioUrl = URL.createObjectURL(audioBlob);

            audioPlayer.src = audioUrl;

 

            const arrayBuffer = await audioBlob.arrayBuffer();

            try {

                const decodingAudioContext = new (window.AudioContext || window.webkitAudioContext)();

                originalAudioBuffer = await decodingAudioContext.decodeAudioData(arrayBuffer);

                durationDisplay.textContent = formatDuration(originalAudioBuffer.duration);

            } catch (e) {

                console.error("Error decoding audio data:", e);

                statusElement.innerHTML = `<strong>Fel:</strong> Kunde inte avkoda ljuddata.`;

                loaderElement.style.display = 'none';

                recordButton.classList.remove('hidden');

                editorElement.classList.add('hidden');

                return;

            }

            loaderElement.style.display = 'none';

            editorElement.classList.remove('hidden');

            recordButton.classList.remove('hidden');

        }

 

        function startTimer() {

            timerInterval = setInterval(() => {

                seconds++;

                timerElement.textContent = formatTime(seconds);

            }, 1000);

        }

 

        function resetTimer() {

            seconds = 0;

            timerElement.textContent = "00:00:00";

            clearInterval(timerInterval);

        }

 

        function formatTime(sec) {

            const h = Math.floor(sec / 3600), m = Math.floor((sec % 3600) / 60), s = sec % 60;

            return [h, m, s].map(v => v.toString().padStart(2, '0')).join(':');

        }

 

        function formatDuration(totalSeconds) {

            const minutes = Math.floor(totalSeconds / 60);

            const seconds = Math.floor(totalSeconds % 60);

            return `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

        }

       

        downloadButton.addEventListener('click', async () => {

            if (!originalAudioBuffer) {

                console.error("No audio buffer to download");

                return;

            }

            const blob = bufferToMp3(originalAudioBuffer);

            download(URL.createObjectURL(blob), `inspelning_angelholm.mp3`);

        });

 

        function download(url, filename) {

            const a = document.createElement('a');

            a.style.display = 'none'; a.href = url; a.download = filename;

            document.body.appendChild(a); a.click(); document.body.removeChild(a);

        }

       

        function bufferToMp3(audioBuffer) {

            const numChannels = audioBuffer.numberOfChannels, sampleRate = audioBuffer.sampleRate, kbps = 128;

            const mp3encoder = new lamejs.Mp3Encoder(numChannels, sampleRate, kbps);

            const pcmData = [];

            for (let i = 0; i < numChannels; i++) pcmData.push(audioBuffer.getChannelData(i));

            const leftChannel = pcmData[0], rightChannel = numChannels > 1 ? pcmData[1] : leftChannel;

            const sampleBlockSize = 1152, mp3Data = [];

            for (let i = 0; i < leftChannel.length; i += sampleBlockSize) {

                const leftChunk = leftChannel.subarray(i, i + sampleBlockSize);

                const rightChunk = rightChannel.subarray(i, i + sampleBlockSize);

                const leftInt16 = new Int16Array(leftChunk.length), rightInt16 = new Int16Array(rightChunk.length);

                for (let j = 0; j < leftChunk.length; j++) {

                    leftInt16[j] = leftChunk[j] * 32767.5 - 0.5;

                    rightInt16[j] = rightChunk[j] * 32767.5 - 0.5;

                }

                const mp3buf = mp3encoder.encodeBuffer(leftInt16, rightInt16);

                if (mp3buf.length > 0) mp3Data.push(new Int8Array(mp3buf));

            }

            const mp3buf = mp3encoder.flush();

            if (mp3buf.length > 0) mp3Data.push(new Int8Array(mp3buf));

            return new Blob(mp3Data, { type: 'audio/mp3' });

        }
